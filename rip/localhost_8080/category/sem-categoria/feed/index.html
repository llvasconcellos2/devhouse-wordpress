SELECT CAST((
    SELECT COLUNA1 + ' - ' COLUNA2 FROM TABELA FOR XML PATH('')
) AS VARCHAR(MAX));

Sendo assim para gerar um JSON de uma query:

SELECT '[' + CAST((
    SELECT '{"coluna1":' + COLUNA1 + '","coluna2":"' + COLUNA2 + '"},' FROM TABELA FOR XML PATH('')
) AS VARCHAR(MAX)) + ']';
