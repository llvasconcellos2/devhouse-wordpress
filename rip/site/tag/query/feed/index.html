SELECT DISTINCT 
    TABLE_NAME AS 'tabela',
    COLUMN_NAME AS 'coluna'
FROM 
    INFORMATION_SCHEMA.COLUMNS
WHERE 
    COLUMN_NAME LIKE '%parte_do_nome_da_coluna%'
AND 
    TABLE_SCHEMA='NOME_DO_BANCO';

USE NOME_DO_BANCO;

SELECT 
    t.name AS 'tabela',
    SCHEMA_NAME(schema_id) AS 'schema',
    c.name AS 'coluna'
FROM 
    sys.tables AS t
INNER JOIN 
    sys.columns c ON t.OBJECT_ID = c.OBJECT_ID
WHERE 
    c.name LIKE '%parte_do_nome_da_coluna%'
ORDER BY 
    schema_id, t.name;
