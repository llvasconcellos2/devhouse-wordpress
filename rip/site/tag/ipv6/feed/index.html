echo net.ipv6.conf.all.disable_ipv6=1 > /etc/sysctl.d/disableipv6.conf
# ou
sysctl net.ipv6.conf.default.disable_ipv6

Esse comando vai desabilitar o ipv6 no próximo boot. Será necessário também comentar as entradas ipv6 do arquivo /etc/hosts.
